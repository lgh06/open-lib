## create token.js
at src/js/token.js
```javascript
const token = {
  'X-LC-Id': '',
  'X-LC-Key': ''
}

export default token;
export {token};
```

``` bash
# install dependencies
npm install

# serve with hot reload at localhost:8080
npm run dev

# build for production with minification
npm run build
```
